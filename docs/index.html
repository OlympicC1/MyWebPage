/** * Simple function to switch between the two "pages" (sections) * in this single-file application. */ function showPage(pageName) { const storyPage = document.getElementById('story-page'); const factsPage = document.getElementById('facts-page'); const storyLink = document.getElementById('story-link'); const factsLink = document.getElementById('facts-link'); if (pageName === 'story') { storyPage.classList.remove('hidden'); factsPage.classList.add('hidden'); storyLink.classList.add('bg-white/10'); factsLink.classList.remove('bg-white/10'); document.title = "Dude: Story & Pics"; } else if (pageName === 'facts') { storyPage.classList.add('hidden'); factsPage.classList.remove('hidden'); storyLink.classList.remove('bg-white/10'); factsLink.classList.add('bg-white/10'); document.title = "Dude: Facts & Safety"; } // Scroll to top of the content area for a clean view transition document.getElementById('top').scrollIntoView(); } // Initialize the view on load window.onload = () => { showPage('story'); };
